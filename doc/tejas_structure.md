// backend/src/cron/statusUpdater.ts

// Runs every 6 hours
cron.schedule("0 */6 * * *", async () => {
  await updateAllApplicationStatuses();
});

async function updateAllApplicationStatuses() {
  const applications = await Application.find({
    status: { $nin: ["rejected", "accepted", "withdrawn", "archived"] },
  });

  for (const app of applications) {
    const user = await User.findById(app.userId);
    const settings = user.settings;

    const daysSinceApplied = Math.floor(
      (Date.now() - app.appliedDate.getTime()) / (1000 * 60 * 60 * 24)
    );

    const daysSinceLastUpdate = Math.floor(
      (Date.now() - app.lastUpdatedAt.getTime()) / (1000 * 60 * 60 * 24)
    );

    // Check for follow-up suggestion
    if (
      app.status === "applied" &&
      daysSinceLastUpdate >= settings.followUpSuggestionDays
    ) {
      app.status = "applied_follow_up";
      await sendNotification(user, "follow_up_suggestion", app);
    }

    // Check for "No Longer Considering"
    if (
      ["applied", "applied_follow_up", "screening"].includes(app.status) &&
      daysSinceLastUpdate >= settings.noLongerConsideringDays
    ) {
      app.status = "no_longer_considering";
      await sendNotification(user, "no_longer_considering", app);
    }

    // Check for auto-archive
    if (
      daysSinceLastUpdate >= settings.autoArchiveDays &&
      app.status === "no_longer_considering"
    ) {
      app.status = "archived";
    }

    // Check interviews awaiting response
    const interview = await Interview.findOne({
      applicationId: app._id,
      status: "completed",
    }).sort({ completedAt: -1 });

    if (interview) {
      const daysSinceInterview = Math.floor(
        (Date.now() - interview.completedAt.getTime()) / (1000 * 60 * 60 * 24)
      );

      if (daysSinceInterview >= settings.postInterviewReminderDays) {
        app.status = "interview_awaiting";
        await sendNotification(user, "interview_awaiting", app);
      }

      if (daysSinceInterview >= 21) {
        app.status = "likely_rejected";
      }
    }

    // Check overdue assessments
    const assessment = await Assessment.findOne({
      applicationId: app._id,
      status: { $in: ["pending", "in_progress"] },
      deadline: { $lt: new Date() },
    });

    if (assessment) {
      const daysOverdue = Math.floor(
        (Date.now() - assessment.deadline.getTime()) / (1000 * 60 * 60 * 24)
      );

      if (daysOverdue > settings.assessmentOverdueGraceDays) {
        assessment.status = "overdue";
        app.status = "assessment_overdue";
        await assessment.save();

        if (daysOverdue >= 7) {
          app.status = "likely_rejected";
        }
      }
    }

    // Check expiring offers
    const offer = await Application.findOne({
      _id: app._id,
      status: "offer_received",
      "offerDetails.expiresAt": {
        $lte: new Date(
          Date.now() + settings.offerExpirationWarningDays * 24 * 60 * 60 * 1000
        ),
      },
    });

    if (offer) {
      app.status = "offer_expiring";
      await sendNotification(user, "offer_expiring", app);
    }

    app.autoStatusLastChecked = new Date();
    await app.save();
  }
}
